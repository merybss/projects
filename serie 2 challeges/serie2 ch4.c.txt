#include <stdio.h>
#include <stdlib.h>
#include<math.h>


int main()
{
    int a,b,c;
    float x1,x2,x;
    double delta;
    printf("entrer les trois valeur a,b,c\n");
    scanf("%d %d %d",&a,&b,&c);
    delta=pow(b,2)-(4*a*c);
    printf("delta=%2.f",delta);
    x1=-b-sqrt(delta)/2*a;
    x2=-b+sqrt(delta)/2*a;
    x=-b/2*a;
    if(delta>0){
     printf("Deux solutions reelles et distinctes:x1=%2.f et x2=%2.f\n",x1,x2);
    }else if(delta=0){
        printf("Une solution double:x=%2.f \n",x);
    }else{
        printf("Pas de solution reelle \n");
    }


    return 0;
}
