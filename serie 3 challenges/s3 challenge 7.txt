#include <stdio.h>
#include <stdlib.h>

int main() {
    int nombre, inverse = 0, reste;

    printf("Entrer un nombre entier : ");
    scanf("%d", &nombre);

    while (nombre != 0) {
        reste = nombre % 10;           
        inverse = inverse * 10 + reste; 
        nombre = nombre / 10;          
    }

    printf("Le nombre inverse est : %d\n", inverse);

    return 0;
}
