#include <stdio.h>
#include <stdlib.h>

int main()
{
    int i,n,min;
    printf("combien d'element voulez vous?: \n");
    scanf("%d",&n);
    int t[i];
    for(i=0;i<n;i++){
        printf("entrer l'element %d\n",i+1);
        scanf("%d",&t[i]);
    }
    min=t[0];
    for(i=0;i<n;i++){
       if(t[i]<min)
        min=t[i];
    }
    printf("l'element minimum est %d\n",min);
    return 0;
}
