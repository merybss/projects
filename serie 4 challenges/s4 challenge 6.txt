#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, i;

    
    printf("Combien d'elements voulez-vous entrer ? : ");
    scanf("%d", &n);

    int t[n], multiplication[n]; 

    for(i = 0; i < n; i++){
        printf("Entrer l'element %d : ", i+1);
        scanf("%d", &t[i]);
    }

    printf("Entrer le facteur de multiplication : ");
    scanf("%d", &m);

    for(i = 0; i < n; i++){
        multiplication[i] = t[i] * m;
    }

    printf("Le tableau final est :\n");
    for(i = 0; i < n; i++){
        printf("%d ", multiplication[i]);
    }
    printf("\n");

    return 0;
}
