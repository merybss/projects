#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>
int main()
{
    char a[50];
    int i,longeur;
    char n;
    int compteur=0;
    printf("saisir la chaine de caractere a: \n");
    fgets(a,sizeof(a),stdin);

    a[strcspn(a, "\n")]='\0';

    printf("entrer l'element a recherche: ");
    scanf("%c",&n);

    longeur=strlen(a);

    for(i=0;i<longeur;i++){
        if(a[i]==n){
       compteur++;

        }
    }
    printf("le nombre de %c est %d",n,compteur);
    return 0;
}
